function exportExcel() {     const p = computeProjection();     const wb = XLSX.utils.book_new();     const makeSheet = (subtitle, rows, totalRows = [], percentRows = []) => {         const data = [[state.setup.businessName], [subtitle], rows[0], ...rows.slice(1)];         const ws = XLSX.utils.aoa_to_sheet(data);         const lastCol = Math.max(1, data[2].length - 1);         ws['!merges'] = [{ s: { r: 0, c: 0 }, e: { r: 0, c: lastCol } }, { s: { r: 1, c: 0 }, e: { r: 1, c: lastCol } }];         applySheetStyles(ws, totalRows.map(r => r + 2), percentRows.map(r => r + 2));         return ws;     };     // 1. Detailed Startup Budget (Newly Added)     const startupDetailRows = [['Item Description', 'Amount (ZAR)'],          ...state.startupCosts.map(c => [c.name, n(c.amount)]),         ['TOTAL STARTUP CAPITAL REQUIRED', p.startupTotal]];     // 2. Sales Forecast (Including Other Income)     const salesForecastRows = [['Line Item', ...p.periods.map(x => x.label)],         ...p.revenueRows.map(row => [row.name, ...p.periods.map((period, idx) => getOverride(row.key, period.key, row.values[idx]))]),         ['Other Income', ...p.periods.map((period, idx) => getOverride(p.otherIncomeRow.key, period.key, p.otherIncomeRow.values[idx]))],         ['Total Sales Revenue', ...p.totalRevenue]];     // 3. Detailed Operating Budget (Salaries + Expenses)     const expenseBudgetRows = [['Line Item', ...p.periods.map(x => x.label)],         ['-- SALARIES --', ...p.periods.map(() => '')],         ...p.salaryRows.map(row => [row.name, ...p.periods.map((period, idx) => getOverride(row.key, period.key, row.values[idx]))]),         ['-- OPERATING EXPENSES --', ...p.periods.map(() => '')],         ...p.expenseRows.map(row => [row.name, ...p.periods.map((period, idx) => getOverride(row.key, period.key, row.values[idx]))]),         ['TOTAL OPERATING EXPENDITURE', ...p.totalExpenses.map((v, i) => v + p.totalSalaries[i])]];     // 4. Income Statement     const incomeRows = [['Line Item', ...p.periods.map(x => x.label)],         ['Sales Revenue', ...p.totalRevenue],         ['Cost of Sales', ...p.totalCOS],         ['Gross Profit', ...p.grossProfit],         ['Operating Expenses (incl. Salaries)', ...p.totalExpenses.map((v, i) => v + p.totalSalaries[i])],         ['Operating Profit (EBIT)', ...p.operatingProfit],         ['Finance Costs', ...p.financeCosts],         ['Net Profit Before Tax', ...p.netProfitBeforeTax],         ['Income Tax', ...p.incomeTax],         ['Net Profit After Tax', ...p.netProfitAfterTax]];     XLSX.utils.book_append_sheet(wb, makeSheet('Startup Budget', startupDetailRows, [startupDetailRows.length - 1]), 'Startup Costs');     XLSX.utils.book_append_sheet(wb, makeSheet('Sales Forecast', salesForecastRows, [salesForecastRows.length - 1]), 'Sales Forecast');     XLSX.utils.book_append_sheet(wb, makeSheet('Expense Budget', expenseBudgetRows, [expenseBudgetRows.length - 1]), 'Expense Budget');     XLSX.utils.book_append_sheet(wb, makeSheet('Income Statement', incomeRows, [3, 5, 9]), 'Income Statement');     // ... [Add Cash Flow and Balance Sheet as per your existing code]          XLSX.writeFile(wb, `${state.setup.businessName.replace(/\s+/g, '_')}_Full_Projections.xlsx`); }
